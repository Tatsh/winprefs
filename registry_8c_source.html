41static inline bool create_dir_recursive(wchar_t *path) {
42 wchar_t folder[MAX_PATH];
43 wchar_t *end;
44 wmemset(folder, L'\0', MAX_PATH);
45 end = wcschr(path, L'\\');
46 while (end != NULL) {
47 wcsncpy(folder, path, (size_t)(end - path + 1));
48 if (!create_dir_wrapper(folder)) {
49 return false;

52 }

109bool export_single_value(const wchar_t *reg_path, HKEY top_key) {
110 wchar_t *first_backslash = wcschr(reg_path, L'\\');
111 if (!first_backslash) {
112 return false;
113 }
114 wchar_t *subkey = first_backslash + 1;
115 HKEY starting_key = HKEY_CURRENT_USER;
116 wchar_t *last_backslash = wcsrchr(reg_path, '\\');
117 wchar_t *value_name_p = last_backslash + 1;
118 size_t value_name_len = wcslen(value_name_p);
119 wchar_t *value_name = calloc(value_name_len, WL);
120 if (!value_name) {
121 return false;
122 }
123 wmemcpy(value_name, value_name_p, value_name_len);
124 *last_backslash = L'\0';
125 if (RegOpenKeyEx(top_key, subkey, 0, KEY_READ, &starting_key) != ERROR_SUCCESS) {
126 free(value_name);
127 debug_print(L"Invalid subkey: '%ls'.\n", subkey);
128 return false;
129 }
130 size_t buf_size = 8192;
131 char *data = malloc(buf_size);
132 DWORD reg_type = REG_NONE;
133 LSTATUS ret = RegQueryValueEx(
134 starting_key, value_name, NULL, &reg_type, (LPBYTE)data, (LPDWORD)&buf_size);
135 if (ret == ERROR_MORE_DATA) {
136 free(data);
137 debug_print(L"Value too large (%ls\\%ls).\n", subkey, value_name);
138 return false;
139 }
140 if (ret != ERROR_SUCCESS) {
141 free(data);
142 debug_print(L"Invalid value name '%ls'.\n", value_name);
143 return false;
144 }
145 if (!do_write_reg_command(
146 GetStdHandle(STD_OUTPUT_HANDLE), reg_path, value_name, data, buf_size, reg_type)) {
147 return false;
148 }
149 free(data);
150 return true;
151}
