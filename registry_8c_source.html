18static inline bool create_dir_recursive(wchar_t *path) {
19 wchar_t folder[MAX_PATH];
20 wchar_t *end;
21 wmemset(folder, L'\0', MAX_PATH);
22 end = wcschr(path, L'\\');
23 while (end != nullptr) {
24 wcsncpy(folder, path, (size_t)(end - path + 1));
25 if (!create_dir_wrapper(folder)) {

29 }
30 if (!create_dir_wrapper(path)) {
31 return false;
32 }
