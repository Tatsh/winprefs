22 if (!has_git()) {

25 }
26 debug_print(L"Committing changes.\n");
27 size_t work_tree_arg_len = wcslen(output_dir) + wcslen(L"--work-tree=") + 1;

33 _snwprintf(work_tree_arg, work_tree_arg_len, L"--work-tree=%ls", output_dir);

47 if (!_wgetcwd(cwd, MAX_PATH)) {
48 return false;
49 }
50 if (_wchdir(output_dir) != 0) {
51 return false;
52 }
53 if (_wspawnlp(P_WAIT, L"git.exe", L"git", L"init", L"--quiet", nullptr) != 0) {

178// Based on https://stackoverflow.com/a/35658917/374110
179wchar_t *get_git_branch(const wchar_t *git_dir_arg,
180 size_t git_dir_arg_len,
181 const wchar_t *work_tree_arg,
182 size_t work_tree_arg_len) {
183 char *result = malloc(255);
184 if (!result) {
185 return nullptr;
186 }
187 memset(result, 0, 255);
188 HANDLE pipe_read, pipe_write;
189 SECURITY_ATTRIBUTES sa_attr = {.lpSecurityDescriptor = nullptr,
190 .bInheritHandle =
191 TRUE, // Pipe handles are inherited by child process.
192 .nLength = sizeof(SECURITY_ATTRIBUTES)};
193 // Create a pipe to get results from child's stdout.
194 if (!CreatePipe(&pipe_read, &pipe_write, &sa_attr, 0)) {
195 free(result);
196 return nullptr;
197 }
198 STARTUPINFOW si = {.dwFlags = STARTF_USESHOWWINDOW | STARTF_USESTDHANDLES,
199 .hStdOutput = pipe_write,
200 .hStdError = pipe_write,
201 .wShowWindow = SW_HIDE};
202 PROCESS_INFORMATION pi = {0};
203 size_t cmd_len = git_dir_arg_len + work_tree_arg_len + 30;
204 wchar_t *cmd = calloc(cmd_len, WL);
205 if (!cmd) {
206 free(result);
207 CloseHandle(pipe_write);
208 CloseHandle(pipe_read);
209 return nullptr;
210 }
211 wmemset(cmd, L'\0', cmd_len);
212 _snwprintf(cmd, cmd_len, L"git.exe %ls %ls branch --show-current", git_dir_arg, work_tree_arg);
213 cmd[cmd_len - 1] = L'\0';
214 BOOL ret = CreateProcess(
215 nullptr, cmd, nullptr, nullptr, TRUE, CREATE_NEW_CONSOLE, nullptr, nullptr, &si, &pi);
216 if (!ret) {
217 CloseHandle(pipe_write);
218 CloseHandle(pipe_read);
219 free(result);
220 return nullptr;
221 }
222 free(cmd);
223 bool proc_ended = false;
224 for (; !proc_ended;) {
225 // Give some time slice (50 ms), so we won't waste 100% CPU.
226 proc_ended = WaitForSingleObject(pi.hProcess, 50) == WAIT_OBJECT_0;
227 // Even if process exited - we continue reading, if
228 // there is some data available over pipe.
229 for (;;) {
230 char buf[255];
231 memset(buf, L'\0', 255);
232 DWORD read = 0;
233 DWORD avail = 0;
234 if (!PeekNamedPipe(pipe_read, nullptr, 0, nullptr, &avail, nullptr)) {
235 break;
236 }
237 if (!avail) { // No data available, return
238 break;
239 }
240 if (!ReadFile(pipe_read, buf, min(sizeof(buf) - 1, avail), &read, nullptr) || !read) {
241 // Error, the child process might have ended
242 break;
243 }
244 buf[min(sizeof(buf) - 1, avail)] = L'\0';
245 if (avail) {
246 strncat(result, buf, proc_ended ? avail - 1 : avail); // Strip newline
247 }
248 }
249 }
250 CloseHandle(pipe_write);
251 CloseHandle(pipe_read);
252 CloseHandle(pi.hProcess);
253 CloseHandle(pi.hThread);
254 size_t res_len = strlen(result);
255 int w_len = MultiByteToWideChar(CP_UTF8, 0, result, (int)res_len, nullptr, 0);
256 wchar_t *w_result = calloc((size_t)w_len + 1, WL);
257 if (!w_result) {
258 return nullptr;
259 }
260 MultiByteToWideChar(CP_UTF8, 0, result, (int)res_len, w_result, w_len);
261 w_result[w_len] = L'\0';
262 free(result);
263 return w_result;
264}
