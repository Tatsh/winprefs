21wchar_t *fix_v_param(const wchar_t *prop, size_t prop_len, bool *heap) {
22 (void)prop_len;
23 *heap = true;
24 if (!wcsncmp(L"(default)", prop, 9) || prop_len == 0 || prop == nullptr) {
25 *heap = false;
26 return REG_PARAM_SLASH_VE;
27 }
28 wchar_t *escaped = escape_for_batch(prop, prop_len);
29 if (escaped == nullptr) {
30 *heap = false;
31 return REG_PARAM_SLASH_V_EMPTY;
32 }
33 size_t escaped_len = (7 + wcslen(escaped)) * WL;
34 wchar_t *out = calloc(7 + wcslen(escaped), WL);
35 if (!out) {
36 return nullptr;
37 }
38 memset(out, 0, escaped_len);
39 _snwprintf(out, escaped_len, L"/v \"%ls\" ", escaped);
40 free(escaped);
41 return out;
42}

44wchar_t *convert_data_for_reg(DWORD reg_type, const char *data, size_t data_len) {
45 if (reg_type == REG_BINARY) {
46 size_t i;
47 size_t n_bin_chars = 2 * data_len;
48 size_t new_len = n_bin_chars + 1;
49 wchar_t *bin = calloc(new_len, WL);
50 if (!bin) {
51 return nullptr;
52 }
53 wmemset(bin, L'\0', new_len);
54 for (i = 0; i < data_len; i++) {
55 wchar_t conv[3];
56 _snwprintf(conv, 3, L"%02x", data[i]);
57 conv[2] = L'\0';
58 wcsncat(&bin[i], conv, 2);
59 }
60 size_t s_size = new_len + 5;
61 wchar_t *out = calloc(s_size, WL);
62 if (!out) {
63 return nullptr;
64 }
65 wmemset(out, L'\0', s_size);
66 _snwprintf(out, s_size, L" /d %ls ", bin);
67 free(bin);
68 return out;
69 }
70 if (reg_type == REG_EXPAND_SZ || reg_type == REG_SZ || reg_type == REG_MULTI_SZ) {
71 wchar_t *s = escape_for_batch((wchar_t *)data, data_len == 0 ? 0 : (data_len / WL) - 1);
72 if (s == nullptr) {
73 return nullptr;
74 }
75 size_t s_size = (wcslen(s) + 8);
76 wchar_t *out = calloc(s_size, WL);
77 if (!out) {
78 return nullptr;
79 }
80 memset(out, 0, s_size);
81 _snwprintf(out, s_size, L" /d \"%ls\" ", s);
82 free(s);
83 return out;
84 }
85 if (reg_type == REG_DWORD || reg_type == REG_QWORD) {
86 size_t s_size = 20;
87 wchar_t *out = calloc(s_size, WL);
88 if (!out) {
89 return nullptr;
90 }
91 memset(out, 0, s_size);
92 if (reg_type == REG_DWORD) {
93 _snwprintf(out, s_size, L" /d %lu ", *(DWORD *)data);
94 } else {
95 _snwprintf(out, s_size, L" /d %llu ", *(unsigned __int64 *)data);
96 }
97 return out;
98 }
99 errno = EINVAL;
100 return nullptr;
101}

103bool do_write_reg_command(HANDLE out_fp,
104 const wchar_t *full_path,
105 const wchar_t *prop,
106 const char *value,
107 size_t data_len,
108 unsigned long type) {
109 wchar_t *escaped_d = convert_data_for_reg(type, value, data_len);
110 wchar_t *escaped_reg_key = escape_for_batch(full_path, wcslen(full_path));
111 bool v_heap = false;
112 wchar_t *v_param = fix_v_param(prop, wcslen(prop), &v_heap);
113 wchar_t reg_type[14];
114 memset(reg_type, 0, sizeof(reg_type));
115 switch (type) {
116 case REG_NONE:
117 wcsncpy(reg_type, L"REG_NONE", 8);
118 break;
119 case REG_BINARY:
120 wcsncpy(reg_type, L"REG_BINARY", 10);
121 break;
122 case REG_SZ:
123 wcsncpy(reg_type, L"REG_SZ", 6);
124 break;
125 case REG_EXPAND_SZ:
126 wcsncpy(reg_type, L"REG_EXPAND_SZ", 13);
127 break;
128 case REG_MULTI_SZ:
129 wcsncpy(reg_type, L"REG_MULTI_SZ", 12);
130 break;
131 case REG_DWORD:
132 wcsncpy(reg_type, L"REG_DWORD", 9);
133 break;
134 case REG_QWORD:
135 wcsncpy(reg_type, L"REG_QWORD", 9);
136 break;
137 }
138 wchar_t *out = calloc(CMD_MAX_COMMAND_LENGTH, WL);
139 if (!out) {
140 return false;
141 }
142 wmemset(out, L'\0', CMD_MAX_COMMAND_LENGTH);
143 int wrote = _snwprintf(out,
144 CMD_MAX_COMMAND_LENGTH,
145 L"reg add \"%ls\" %ls/t %ls%ls/f",
146 escaped_reg_key,
147 v_param,
148 reg_type,
149 escaped_d ? escaped_d : L" ");
150 if (((size_t)wrote < CMD_MAX_COMMAND_LENGTH) ||
151 ((size_t)wrote == CMD_MAX_COMMAND_LENGTH && out[CMD_MAX_COMMAND_LENGTH - 1] == L'f' &&
152 out[CMD_MAX_COMMAND_LENGTH - 2] == L'/' && out[CMD_MAX_COMMAND_LENGTH - 3] == L' ')) {
153 size_t req_size =
154 (size_t)WideCharToMultiByte(CP_UTF8,
155 IsWindowsVistaOrGreater() ? _WC_ERR_INVALID_CHARS : 0,
156 out,
157 -1,
158 NULL,
159 0,
160 NULL,
161 NULL);
162 char *mb_out = malloc(req_size);
163 WideCharToMultiByte(CP_UTF8,
164 IsWindowsVistaOrGreater() ? _WC_ERR_INVALID_CHARS : 0,
165 out,
166 -1,
167 mb_out,
168 (int)req_size,
169 NULL,
170 NULL);
171 mb_out[req_size - 1] = '\n';
172 DWORD written;
173 WriteFile(out_fp, mb_out, (DWORD)(req_size), &written, nullptr);
174 free(mb_out);
175 } else {
176 debug_print(L"%ls %ls: Skipping due to length of command.\n", full_path, prop);
177 }
178 if (escaped_d) {
179 free(escaped_d);
180 }
181 if (v_param && v_heap) {
182 free(v_param);
183 }
184 free(out);
185 free(escaped_reg_key);
186 return true;
187}

189bool do_write_reg_commands(HANDLE out_fp, HKEY hk, unsigned n_values, const wchar_t *full_path) {
190 if (n_values == 0) {
191 return true;
192 }
193 if (!out_fp || !full_path) {
194 errno = EINVAL;
195 return false;
196 }
197 size_t data_len;
198 DWORD i;
199 DWORD reg_type;
200 size_t value_len;
201 wchar_t *value = calloc(MAX_VALUE_NAME, WL);
202 if (!value) {
203 return false;
204 }
205 int ret = ERROR_SUCCESS;
206 char data[8192];
207 for (i = 0; i < n_values; i++) {
208 data_len = sizeof(data);
209 wmemset(value, L'\0', MAX_VALUE_NAME);
210 value_len = MAX_VALUE_NAME * WL;
211 reg_type = REG_NONE;
212 ret = RegEnumValue(
213 hk, i, value, (LPDWORD)&value_len, 0, &reg_type, (LPBYTE)data, (LPDWORD)&data_len);
214 if (ret == ERROR_MORE_DATA) {
215 continue;
216 }
217 if (ret == ERROR_NO_MORE_ITEMS) {
218 break;
219 }
220 do_write_reg_command(out_fp, full_path, value, data, data_len, reg_type);
221 }
222 free(value);
223 return true;
224}

300 } else {
301 debug_print(L"%ls: Skipping %ls because RegEnumKeyEx() failed.\n",
302 prior_stem,
303 full_path);
304 }
305 }
306 free(ach_key);
307 } else {
308 debug_print(L"%ls: No subkeys in %ls.\n", prior_stem, stem);
309 }
310 if (n_values) {
311 if (!do_write_reg_commands(out_fp, hk_out, n_values, full_path)) {
312 return false;
313 }
314 } else {
315 debug_print(L"%ls: No values in %ls.\n", prior_stem, stem);
316 }
317 RegCloseKey(hk_out);
318 } else {
319 debug_print(L"%ls: Skipping %ls. Does the location exist?\n", prior_stem, stem);
320 }
321 free(full_path);
322 return true;
